  <body>
    <h2>画像ファイル名変換・ダウンロードツール</h2>

    <label>
      出力形式：
      <select id="format">
        <option value="jpg">JPG</option>
        <option value="png">PNG</option>
        <option value="webp">WEBP</option>
      </select>
    </label>

    <p>
      行ごとにコードを入力し、横に最大5枚の画像をドラッグ＆ドロップしてください。
    </p>

    <table id="table">
      <tr>
        <th>コード</th>
        <th>画像1</th>
        <th>画像2</th>
        <th>画像3</th>
        <th>画像4</th>
        <th>画像5</th>
      </tr>
    </table>

    <button onclick="downloadAllImages()">画像ダウンロード</button>
    <button onclick="resetAll()">リセット</button>

    <script>
      const table = document.getElementById("table");
      const rows = 10;
      const cols = 5;
      const images = [];

      // テーブル作成
      for (let r = 0; r < rows; r++) {
        images[r] = [];
        const tr = document.createElement("tr");

        // SKU入力
        const nameTd = document.createElement("td");
        const nameInput = document.createElement("input");
        nameInput.id = "name" + r;
        nameTd.appendChild(nameInput);
        tr.appendChild(nameTd);

        // 画像列
        for (let c = 0; c < cols; c++) {
          const td = document.createElement("td");
          const drop = document.createElement("div");
          drop.className = "drop";
          drop.innerText = "DROP";
          const img = document.createElement("img");
          const fileInput = document.createElement("input");
          fileInput.type = "file";
          fileInput.accept = "image/*";
          fileInput.style.display = "none";

          // クリックで選択
          drop.onclick = () => fileInput.click();
          fileInput.onchange = (e) => {
            loadFile(e.target.files[0], r, c, img, drop);
          };

          // ドラッグ＆ドロップ
          drop.ondragover = (e) => e.preventDefault();
          drop.ondrop = (e) => {
            e.preventDefault();
            const files = e.dataTransfer.files;
            if (files.length > 0) {
              loadFile(e.dataTransfer.files[0], r, c, img, drop);
            }
          };

          td.appendChild(drop);
          td.appendChild(fileInput);
          tr.appendChild(td);
        }
        table.appendChild(tr);
      }

      // 画像読み込みとプレビュー
      function loadFile(file, r, c, img, drop) {
        images[r][c] = file;
        const reader = new FileReader();
        reader.onload = (e) => {
          img.src = e.target.result;
          drop.innerHTML = "";
          drop.appendChild(img);
        };
        reader.readAsDataURL(file);
      }

      // 画像変換（JPG元画質保持）
      function convertImage(file, format) {
        return new Promise((resolve) => {
          if (format === "jpg" && file.type === "image/jpeg") {
            resolve(file);
            return;
          }
          const img = new Image();
          const reader = new FileReader();
          reader.onload = (e) => (img.src = e.target.result);
          img.onload = () => {
            const canvas = document.createElement("canvas");
            canvas.width = img.width;
            canvas.height = img.height;
            const ctx = canvas.getContext("2d");
            ctx.drawImage(img, 0, 0);
            let type = "image/jpeg";
            if (format === "png") type = "image/png";
            if (format === "webp") type = "image/webp";
            canvas.toBlob((blob) => resolve(blob), type, 1.0);
          };
          reader.readAsDataURL(file);
        });
      }

      // 画像個別ダウンロード
      async function downloadAllImages() {
        const format = document.getElementById("format").value;
        for (let r = 0; r < rows; r++) {
          const name = document.getElementById("name" + r).value;
          if (!name) continue;
          for (let c = 0; c < cols; c++) {
            const file = images[r][c];
            if (!file) continue;
            const imgBlob = await convertImage(file, format);
            const filename = name + "_" + (c + 1) + "." + format;
            const a = document.createElement("a");
            a.href = URL.createObjectURL(imgBlob);
            a.download = filename;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
          }
        }
      }

      // 入力リセット
      function resetAll() {
        for (let r = 0; r < rows; r++) {
          document.getElementById("name" + r).value = "";
          for (let c = 0; c < cols; c++) {
            images[r][c] = null;
            const td = table.rows[r + 1].cells[c + 1];
            const drop = td.querySelector(".drop");
            drop.innerHTML = "DROP";
          }
        }
      }
    </script>
